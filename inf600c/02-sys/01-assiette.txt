#!/usr/bin/python3

import os

cwd = os.path.dirname(os.path.realpath(__file__))
plats = os.listdir(os.path.join(cwd, 'plats'))
plats.sort()

print("Pardons, j'ai mal entendu. Qu'est-ce que vous vouliez que je vous passe?")
for p in plats:
    if '.' not in p:
        print(p)

choix = input().strip()

if choix in plats:
    if 'flag' in choix:
        print("Arthour! Cuillere!")
    else:
        filename = os.path.join("plats/" + choix)
        with open(filename, 'r') as fd:
            print('\n' + fd.read())
else:
    print("Arthour! Cuillere!")

- il y un chemin relatif pour aller chercher le plats 
- creer un dossier du nom plats dans son home directory
- lien symbolique entre le flag et un fichier fruits ou fromage ou crudites (un present dans les choix) -> ln -s /quetes/02-sys/01-assiete/flag fruit
- l'executer l'executable dans son home directory (/quetes/02-sys/01-assiete/assiette) el il va chercher l'element dans notre repertoire car chemin relatif
